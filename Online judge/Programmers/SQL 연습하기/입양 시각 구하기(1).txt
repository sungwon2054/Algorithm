SELECT SUBSTRING(DATETIME, 12, 2),COUNT(SUBSTRING(DATETIME, 12, 2))
FROM ANIMAL_OUTS 
WHERE SUBSTRING(DATETIME, 12, 2) > 8 AND SUBSTRING(DATETIME, 12, 2) < 20
GROUP BY SUBSTRING(DATETIME, 12, 2)
ORDER BY SUBSTRING(DATETIME, 12, 2)

// 시간만 추출해주는 HOUR 함수 이용
SELECT HOUR(DATETIME) AS HOUR, COUNT(DATETIME) AS COUNT
FROM ANIMAL_OUTS
WHERE HOUR(DATETIME) >= 9
  AND HOUR(DATETIME) <= 19
GROUP BY HOUR(DATETIME)
